#include <stdio.h>
#include<string.h>

struct output
{
    int count;
    char *str;
};
int main()
{
    struct output out[20];
    int i=0;
    char sub[5];
    char sub1[5];
    int j=0;
    int k=0;
    int l=0;
    int flag=0;
    int flag1=0;
    int max=0;
    int counter=0;
    int n=0;
    char input[100]="hello red hello green hello yellow";
    while(1)
    {
        k=0;
        l=0;
        flag=0;
        
        for(j=i;j<(i+4);j++)
        {
            if(input[j]=='\0')
            {
                flag=1;
                break;
            }
            sub[k]=input[j];
            k++;
        }
        if(flag)
            break;
        sub[k]='\0';
        k=0;
        j=0;
        flag1=0;
        while(1)
        {
            for(k=j;k<(j+4);k++)
            {
                if(input[k]=='\0')
                {
                    flag1=1;
                    break;
                }
                 
                sub1[l]=input[k];
                l++;
            }
            
            if(flag1)
                break;
            sub1[l]='\0';
            
           
            if(strcmp(sub,sub1)==0)
            {
               // printf("%s  %s\n",sub,sub1);
                counter++;
            }
                
            l=0;
            j++;
        }
        
       
        if(max<=counter)
        {
            if(n!=0)
            {
                printf("%s  %s\n",out[n-1].str,sub);
                if(strcmp(out[n-1].str,sub)!=0)
                {
                   // printf("%d\n",n);
                    max=counter;
                    out[n].count=max;
                    out[n].str=sub;
                    n++;
                 //   printf("%d  %s\n",counter,sub);
                }
            }
            else
            {
              //  printf("%s\n",sub);
                max=counter;
                out[n].count=max;
                out[n].str=sub;
                n++;
            }
        }
        counter=0;
        i++;
    }
   /* printf("%d\n",n);
   for(i=0;i<n;i++)
    {
        printf("%d  %s\n",out[i].count,out[i].str);
    }*/
    return 0;
}
